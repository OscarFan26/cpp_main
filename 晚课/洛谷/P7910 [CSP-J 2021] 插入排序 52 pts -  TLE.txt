#include <bits/stdc++.h>
using namespace std;

int n, q, a[8000];

int sort_x(int record) {
  int b[8000] = {};
  for (int i = 0; i < n; i++) b[i] = a[i];
  for (int i = 1; i < n; i++)
    for (int j = i; j >= 1; j--)
      if (b[j] < b[j - 1]) {
        if (record == j)
          record = j - 1;
        else if (record == j - 1)
          record = j;
        swap(b[j - 1], b[j]);
      }
  return record;
}

int main() {
  cin >> n >> q;
  for (int i = 0; i < n; i++) cin >> a[i];
  for (int i = 0; i < q; i++) {
    char op;
    cin >> op;
    if (op == '1') {
      int x, v;
      cin >> x >> v;
      a[x - 1] = v;
    }
    else if (op == '2') {
      int x;
      cin >> x;
      cout << sort_x(x-1) + 1 << "\n";
    }
  }
  return 0;
}
